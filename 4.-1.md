* Fibonacci 순열은 아래와 같이 정의됨 F0 = 0 F1 = 1 Fn = Fn - 1 + Fn - 2, n &gt;= 2

```python
# Using Recursion

def solution1(x):
    if x == 0:
        answer = 0
    elif x == 1:
        answer = 1

# Using Iterative
# 이해안감

def solution2(x):
    a = 0
    b = 1

    for _ in range(0, x):
        # Disply the current Fibonacci number
        print(a)
        temp = a
        a = b
        b = temp + b
    return a

print(solution2(10))
```
